class Solution {
public:
    int maxArea(vector<int>& height) {
        int n=height.size();
        int maxwater=0;
        int left=0;
        int right=n-1;
        while(left<right){
            int h=min(height[left],height[right]);
            int w= right-left;
            int water=h*w;
            maxwater=max(maxwater,water);
            height[left]<height[right]?left++:right--;
        

        }
        return maxwater;
     

        
    }
};
