class Solution {
public:
    int characterReplacement(string s, int k) {
        int l=0;
        int r=0;
        int maxfreq=0;
        int res=0;
        vector<int>freq(26,0);
        while(r<s.size()){ 
            freq[s[r]-'A']++;
            maxfreq=max(freq[s[r]-'A'],maxfreq);
             while((r-l+1)-maxfreq>k){
            freq[s[l]-'A']--;
            l++;
        }
        res=max(res,r-l+1);
        r++;
        
        }
        return res;
       
    }

};
