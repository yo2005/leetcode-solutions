class Solution {
public:
    int longestConsecutive(vector<int>& nums) {

        unordered_set<int>st;
        for(auto it:nums){
            st.insert(it);
        }
        int longest=0;
        for(auto it:st){
            if(st.find(it-1)==st.end()){
                int currentnum=it;
                int length=1;
                 while(st.find(currentnum+1)!=st.end()){
                currentnum ++;
                length++;

            }
             longest=max(longest,length);


            }
           
          
        }
       return longest;

        
    }
};
