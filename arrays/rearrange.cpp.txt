class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        int n=nums.size();
        int posindex=0;
        int negindex=1;
        vector<int>a(n,0);

        for(int i=0;i<n;i++){
            if(nums[i]>0){
                a[posindex]=nums[i];
                posindex+=2;
            }else{
                a[negindex]=nums[i];
                negindex+=2;

            }

        }
        return a;

        
        
    }
};
