class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int l=0;//buy
        int r=1;//sell
        int maxp=0;
        while(r<prices.size()){
            //profitable?
            if(prices[l]<prices[r]){
                int profit=prices[r]-prices[l];
                maxp=max(maxp,profit);
            }
            else{
                l=r;

            }
            r++;
        }
        return maxp;
     
    }
};
