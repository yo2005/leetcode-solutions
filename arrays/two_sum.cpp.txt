// LeetCode 1: Two Sum
// Difficulty: Easy
// Topic: Arrays, Hashing
// Time Complexity: O(n)
// Space Complexity: O(n)
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int>mp;
        for(int i=0;i<nums.size();i++){
            int a= nums[i];
            int more=target-a;
            if(mp.find(more)!=mp.end()){
                return {mp[more],i};
            }
            mp[a]=i;


        }
        return{-1,-1};
        
    }
};
