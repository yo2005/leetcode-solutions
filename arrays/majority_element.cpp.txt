class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int freq=0;
        int mj;
        for(int i=0;i<nums.size();i++){
            if(freq==0){
                freq=1;
                mj=nums[i];
            }
            else if(mj==nums[i]){
                freq++;
            }
            else{
                freq--;
            }
        }
        return mj;

        
      
     
        
    }
};
